\section{Synthetic Control}

% --- Frame 1: Motivation ------------------------------------
\begin{frame}{The single-treated-unit problem}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{When DiD breaks down}
      \begin{itemize}
        \item DiD needs many treated units for inference
        \item Policy often applies to \emph{one} state, country, or city
        \item Who is the right control group?
        \item Researcher discretion $\Rightarrow$ cherry-picking risk
      \end{itemize}

      \bigskip
      \textbf{The question \citep{abadie2010}}\\[2pt]
      California passes Proposition~99 in 1988 — a large tobacco tax.
      \emph{What would cigarette consumption have been without it?}

      \medskip
      \begin{alertblock}{The challenge}
        No single state is a good control. Using all states or a
        researcher-chosen subset gives very different answers.
      \end{alertblock}
    \end{column}
    \begin{column}{0.44\textwidth}
      % --- TikZ: raw trends sketch ---
      \begin{tikzpicture}[scale=0.85]
        \draw[->] (0,0) -- (5.2,0) node[right]{\scriptsize year};
        \draw[->] (0,0) -- (0,4.2) node[above]{\scriptsize cigarettes};

      \centering\scriptsize
      Post-1988 gap = treatment effect \emph{if} trends were parallel.
      But are they?
    \end{column}
  \end{columns}

\end{frame}

% --- Frame 2: The method ------------------------------------
\begin{frame}{Synthetic Control: building the counterfactual}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Setup} \citep{abadie2010}
      \begin{itemize}
        \item $J+1$ units, $T$ periods; unit~1 treated at $T_0$
        \item \emph{Donor pool}: units $2,\ldots,J+1$ never treated
        \item \emph{Synthetic control}:
          $\hat{Y}_{1t}^{(0)} = \textstyle\sum_{j=2}^{J+1} w_j^{\,*}\, Y_{jt}$
        \item Weights: $w_j^* \geq 0$, $\textstyle\sum w_j^* = 1$
      \end{itemize}

      \medskip
      \textbf{How weights are chosen}\\
      Minimise pre-treatment prediction error:
      \[
        W^* = \arg\min_W \; \|X_1 - X_0 W\|_V
      \]
      $X_1, X_0$ = pre-treatment predictors (outcome lags + covariates);
      $V$ = importance matrix (cross-validated)

      \medskip
      \begin{block}{Key property}
        Weights are \textbf{non-negative and sum to one} $\Rightarrow$
        interpolation only; no extrapolation beyond the convex hull of
        donor units.
      \end{block}
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{California Prop.~99: result}

      \smallskip
      % --- TikZ: gap plot ---
      \begin{tikzpicture}[scale=0.85]
        \draw[->] (0,0) -- (5.2,0) node[right]{\scriptsize year};
        \draw[->] (0,0) -- (0,4.2) node[above]{\scriptsize cigarettes};
        \draw[dashed, ecgray] (2.8,0) -- (2.8,4) node[above]{\scriptsize 1988};

      \smallskip

\end{frame}

% --- Frame 3: Inference -------------------------------------
\begin{frame}{Inference: permutation tests}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{The problem}
      \begin{itemize}
        \item One treated unit $\Rightarrow$ no asymptotic standard errors
        \item Cannot bootstrap (only one treated obs.)
      \end{itemize}

      \medskip
      \textbf{Placebo/permutation approach}
      \begin{enumerate}
        \item Apply the synthetic control method to \emph{each donor state}
              as if it had been treated in 1988
        \item Compute the post/pre RMSPE ratio for each state:\\
          \[
            r_j = \frac{\text{RMSPE}_{\text{post}}}{\text{RMSPE}_{\text{pre}}}
          \]
        \item $p$-value = fraction of placebo $r_j \geq r_{\text{CA}}$
      \end{enumerate}

      \medskip
      \begin{block}{Result}
        CA ranks \textbf{1st} out of 39 states (some excluded for poor pre-fit).
        $p = 1/39 \approx 0.026$ — significant at 5\%.
      \end{block}
    \end{column}
    \begin{column}{0.46\textwidth}
      % --- TikZ: spaghetti gaps ---
      \begin{tikzpicture}[scale=0.85]
        \draw[->] (0,1.5) -- (5.2,1.5) node[right]{\scriptsize year};
        \draw[->] (0,0) -- (0,4.0) node[above]{\scriptsize gap};
        \draw[dashed, ecgray] (2.8,0) -- (2.8,4.0);
        \node[above] at (2.8,4.0) {\scriptsize 1988};

% --- Frame 4: Limitations -----------------------------------
\begin{frame}[shrink=5]{Synthetic Control: when it works — and when it doesn't}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{When synth works well}
        \begin{itemize}
          \item Aggregate-level treatment (state, country, city)
          \item Long pre-treatment period ($T_0 \geq 10$ periods)
          \item Good pre-treatment fit available from donor pool
          \item Single or few treated units
          \item Policy well-defined with known timing
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{When synth fails}
        \begin{itemize}
          \item No good match in donor pool (large pre-RMSPE)
          \item Donor units also partially treated or anticipate treatment
          \item Short pre-period $\Rightarrow$ over-fit
          \item Spillovers from treated to donors
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}

  \medskip
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Synth vs.\ DiD}
      \begin{itemize}\small
        \item Synth: transparent, auditable, $N_\text{treated}=1$
        \item DiD: scalable to many units, easier inference
        \item Both identify ATT; synth makes the counterfactual explicit
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Extensions}
      \begin{itemize}\small
        \item Staggered synth: multiple cohorts \citep{abadie2010}
        \item Synthetic DiD (Arkhangelsky et al.\ 2021): combines unit
              and time weights
        \item `synth` (Stata/R); `scpi` for uncertainty quantification
      \end{itemize}
    \end{column}
  \end{columns}

\end{frame}


% --- Frame 5: Synthetic DiD extension -------------------------
\begin{frame}{Extension: Synthetic Difference-in-Differences}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Bridging Synth and DiD} \citep{arkhangelsky2021}\\[4pt]
      Both methods share a flaw:
      \begin{itemize}
        \item DiD: parallel trends may not hold
        \item Synth: treated unit must lie in the convex hull of donors
      \end{itemize}

      \medskip
      \textbf{SDID solves both by combining}
      \begin{itemize}
        \item \emph{Unit weights} $\hat\omega_j$ — like synthetic control,
              re-weight donors to match treated pre-trends
        \item \emph{Time weights} $\hat\lambda_t$ — up-weight
              pre-treatment periods that look most like the treatment period
      \end{itemize}

      \medskip
      \textbf{Estimator}
      \[
        \hat\tau^{\text{SDID}} =
        \arg\min_{\tau,\alpha,\beta}\,
        \sum_{i,t} \hat\omega_i\,\hat\lambda_t
        \bigl(Y_{it} - \alpha_i - \beta_t - \tau D_{it}\bigr)^2
      \]
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{Key properties}
        \begin{itemize}\small
          \item Consistent even if trends are \emph{not} parallel
          \item Works with multiple treated units (unlike basic synth)
          \item Nests DiD ($\hat\omega_j = 1/N_{\text{ctrl}}$) and SC
                ($\hat\lambda_t = 1/T_{\text{pre}}$)
          \item Bootstrap/placebo inference
        \end{itemize}
      \end{block}

      \medskip
      \textbf{Stata: \texttt{sdid}} \citep{clarke2024sdid}\\[2pt]
      \begin{itemize}\small
        \item Syntax: \texttt{sdid Y id t D, vce(bootstrap)}
        \item Supports covariates, staggered adoption,
              and graphical output
        \item DoE application: \citet{rizzo2025doe} — effect on staff
              recruitment in Excellence-funded departments
      \end{itemize}

      \medskip
      \begin{exampleblock}{When to prefer SDID over Synth}
        \small Many treated units; possible pre-trend divergence;
        want standard errors with a formula (not just permutation).
      \end{exampleblock}
    \end{column}
  \end{columns}

\end{frame}

\section{Regression Discontinuity Design}

% --- Frame 5: The design ------------------------------------
\begin{frame}{Sharp RDD: threshold and local randomisation}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{The design}
      \begin{itemize}
        \item Running variable $X_i$ (score, index, vote margin)
        \item Threshold $c$: treatment $D_i = \mathbf{1}[X_i \geq c]$
        \item Units just above and below $c$ are locally comparable
        \item Assignment is ``as good as random'' near the cutoff
      \end{itemize}

      \medskip
      \textbf{Meyersson (2014) — Turkey} \citep{meyersson2014}\\[2pt]
      \begin{itemize}
        \item 1994 Turkish municipal elections
        \item Running variable: Islamic party vote \emph{margin}
        \item Treatment: Islamic mayor elected ($X_i \geq 0$)
        \item Outcome: female high-school completion rate (2000)
        \item Finding: +4.4~pp for women in Islamic-won municipalities
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      % --- TikZ: classic RDD threshold plot ---
      \begin{tikzpicture}[scale=0.90]
        \draw[->] (-2.6,0) -- (2.6,0) node[right]{\scriptsize $X$ (margin)};
        \draw[->] (0,-0.2) -- (0,4.0) node[above]{\scriptsize $Y$};
        % Threshold
        \draw[dashed, ecgray] (0,-0.2) -- (0,3.8);
        \node[below] at (0,0) {\scriptsize $c=0$};
        % Left branch (control)
        \draw[ecblue, thick, domain=-2.4:0, samples=30]
          plot (\x, {1.2 + 0.4*\x + 0.05*\x*\x});
        % Right branch (treated) — higher intercept
        \draw[ecred, thick, domain=0:2.4, samples=30]
          plot (\x, {2.2 + 0.4*\x + 0.05*\x*\x});
        % Dots scatter suggestion
        \foreach \x/\y in {-2.0/0.5, -1.5/0.8, -1.0/1.0, -0.5/1.1,

          \filldraw[ecgray!70] (\x,\y+0.1*\x) circle (1.2pt);
        }
        % Jump arrow
        \draw[<->, thick, ecred]
          (0.08, {1.2}) -- (0.08, {2.2})
          node[midway, right=2pt]{\scriptsize $\hat\tau$};
        % Labels
        \node[ecblue, left] at (-2.4, {1.2-0.96+0.05*5.76}) {\scriptsize Control};
        \node[ecred, right] at (2.4, {2.2+0.96+0.05*5.76})  {\scriptsize Treated};
      \end{tikzpicture}

      \smallskip\centering\small
      Jump at $c$ = causal effect \emph{if} no other discontinuity there.
    \end{column}
  \end{columns}

\end{frame}

% --- Frame 6: Identification ---------------------------------
\begin{frame}{Identification: the continuity assumption}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Formal assumption}\\[4pt]
      The potential outcome functions
      \[
        E\!\left[Y_i^{(0)}\mid X_i=x\right] \text{ and }
        E\!\left[Y_i^{(1)}\mid X_i=x\right]
      \]
      are \textbf{continuous} in $x$ at $c$.

      \medskip
      \textbf{What this buys}
      \[
        \hat\tau_{\text{RDD}} =
        \lim_{x\downarrow c} E[Y\mid X=x]
        - \lim_{x\uparrow c} E[Y\mid X=x]
      \]
      Identifies the \textbf{Local Average Treatment Effect at $c$}.

      \medskip
      \textbf{What it doesn't}
      \begin{itemize}
        \item LATE $\neq$ ATE (effect may vary away from threshold)
        \item Cannot extrapolate away from $c$
        \item Says nothing about why the discontinuity holds
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{alertblock}{Key violation: manipulation}
        If units can sort above/below the threshold, $X_i$ is
        \textbf{not} continuously distributed at $c$.

        \medskip
        Examples:
        \begin{itemize}\small
          \item Students repeating exams to score just above a grade cutoff
          \item Political parties inflating vote counts near 50\%
          \item Firms manipulating accounting ratios near regulatory thresholds
        \end{itemize}
        \medskip
        Detect with: \texttt{rddensity} \citep{mccrary2008}
      \end{alertblock}

      \medskip
      \begin{block}{Local vs.\ global}
        \small RDD is \emph{local}: only observations near $c$ identify
        $\tau$. This is a feature (credibility) and a limitation
        (external validity).
      \end{block}
    \end{column}
  \end{columns}

\end{frame}

% --- Frame 7: Estimation -------------------------------------
\begin{frame}{Estimation: local linear regression and bandwidth}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Local linear regression (LLR)}\\[4pt]
      Within bandwidth $h$ on each side of $c$, fit:
      \[
        Y_i = \alpha + \tau D_i + \beta(X_i - c) + \gamma D_i(X_i - c) + \varepsilon_i
      \]
      with \textbf{triangular kernel} weights $K\!\left(\tfrac{X_i-c}{h}\right)$.

      \medskip
      \begin{itemize}
        \item LLR preferred over polynomial regression
              (lower bias at boundary)
        \item Triangular kernel preferred over uniform
              (down-weights distant observations)
        \item Two separate regressions or one with interaction
      \end{itemize}

      \medskip
      \textbf{Bandwidth selection \citep{calonico2014}}
      \begin{itemize}
        \item MSE-optimal: $h^* = \arg\min \text{MSE}(\hat\tau)$
        \item Bias-correction + robust CIs (CCT 2014)
        \item \texttt{rdrobust} default: \texttt{bwselect(mserd)}
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Reading \texttt{rdrobust} output}

      \smallskip
      \begin{center}\scriptsize
      \renewcommand{\arraystretch}{1.3}
      \begin{tabular}{ll}
        \toprule
        \texttt{Bandwidth h}     & Optimal $h$ left/right \\
        \texttt{N (left/right)}  & Obs.\ within $h$ \\
        \texttt{Coef.\ Conventional} & $\hat\tau$ (biased) \\
        \texttt{Coef.\ Bias-corrected} & $\hat\tau_{bc}$ \\
        \texttt{CI Robust}       & CCI-robust interval \\
        \bottomrule
      \end{tabular}
      \end{center}

      \medskip
      \begin{exampleblock}{Meyersson result}
        \small\texttt{rdrobust} with MSE-optimal bandwidth:
        \[
          \hat\tau = 0.044\ \ (0.024)
        \]
        +4.4~pp female high-school completion.
        Robust 95\% CI: $[0.001, 0.087]$.
      \end{exampleblock}

      \medskip\small
      \textbf{Rule of thumb:}\\
      Triangular kernel + local linear + CCT bandwidth = standard.
    \end{column}
  \end{columns}

\end{frame}

% --- Frame 8: Validity tests ---------------------------------
\begin{frame}{Validity: what you must always check}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{1.\ Density test (McCrary / \texttt{rddensity})}
        Is the density of $X_i$ continuous at $c$? \\
        A jump $\Rightarrow$ units are sorting; running variable is
        manipulated.\\[4pt]
        \texttt{rddensity score, c(0)}\\
        \emph{H$_0$: no density discontinuity}; want $p>0.05$
      \end{block}

      \medskip
      \begin{block}{2.\ Covariate balance}
        Run RDD on pre-determined covariates.\\
        Should find \textbf{no jump} at $c$ — if they jump, something
        else changes at the threshold.\\[4pt]
        \texttt{rdrobust covariate score, c(0)}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{3.\ Placebo cutoffs}
        Test at arbitrary thresholds $c' \neq c$ within the support.\\
        Should find \textbf{no significant effect} away from the real cutoff.
      \end{block}

      \medskip
      \begin{block}{4.\ Bandwidth sensitivity}
        Plot $\hat\tau(h)$ for a range of bandwidths.\\
        A stable estimate reassures; a wildly varying one warns.
      \end{block}

      \medskip
      \begin{alertblock}{Donut RDD}
        Exclude obs.\ within a small radius $\delta$ of $c$.\\
        Tests whether the effect is driven by near-threshold
        manipulation. Effect should persist with $\delta>0$.
      \end{alertblock}
    \end{column}
  \end{columns}

\end{frame}

\section{Bartik / Shift-Share IV}
% ============================================================

% --- Frame 9: What is SSIV ----------------------------------
\begin{frame}{The shift-share instrument}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{The formula}
      \[
        z_\ell = \sum_{n=1}^{N} \underbrace{s_{\ell n}}_{\text{shares}}
                 \cdot \underbrace{g_n}_{\text{shocks}}
      \]
      \begin{itemize}
        \item $s_{\ell n}$: industry $n$'s share of local labour in region $\ell$
              (lagged, pre-determined)
        \item $g_n$: national (leave-one-out) growth of industry $n$
        \item $z_\ell$: predicted local employment growth from national
              industry trends
      \end{itemize}

      \medskip
      \textbf{Idea \citep{bartik1991}}\\
      Two regions with different industry mixes will receive
      \emph{different shocks} even if the national trend is the same.
      This generates exogenous variation in the endogenous treatment.
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Three canonical applications}

      \smallskip
      \begin{center}\scriptsize
      \renewcommand{\arraystretch}{1.4}
      \begin{tabular}{lll}
        \toprule
        Paper & Shares $s_{\ell n}$ & Shocks $g_n$ \\
        \midrule
        Bartik (1991) & Lagged industry empl.\ & National empl.\ growth \\
        ADH (2013) & Manuf.\ empl.\ shares & China exports to \\
                   & (10-yr lag) & 8 non-US countries \\
        Card (2009) & Migrant enclaves & National immigration \\
                    & (past shares) & by origin country \\
        \bottomrule
      \end{tabular}
      \end{center}

      \smallskip
      \begin{block}{\small Notation}
        \small Model: $y_\ell = \beta x_\ell + \gamma' w_\ell + \varepsilon_\ell$\\
        Use $z_\ell$ as IV for the endogenous $x_\ell$.\\
        Validity: $E[z_\ell \varepsilon_\ell] = 0$
      \end{block}
    \end{column}
  \end{columns}

\end{frame}

% --- Frame 10: Two identification strategies ----------------
\begin{frame}[shrink=5]{Two ways to justify SSIV}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Strategy 1: Shock exogeneity (BHJ)}
        \citet{borusyak2022}: exogeneity lives in the \textbf{shocks} $g_n$.

        \medskip
        \textbf{Assumption:} $g_n$ are as-good-as-randomly assigned
        across industries, conditional on controls.

        \medskip
        \textbf{Implication:}\\
        Shares $s_{\ell n}$ are just heterogeneous \emph{exposure weights},
        not themselves required to be exogenous.

        \medskip
        \textbf{Test:} balance at the shock (industry) level —
        do pre-trends in $g_n$ correlate with industry characteristics?

        \medskip
        \textbf{Inference:} cluster standard errors at the
        \emph{industry} (shock) level.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Strategy 2: Share exogeneity (GPSS)}
        \citet{goldsmithpinkham2020}: exogeneity lives in the
        \textbf{shares} $s_{\ell n}$.

        \medskip
        \textbf{Assumption:} $s_{\ell n}$ pre-determined and uncorrelated
        with local unobservables $\varepsilon_\ell$.

        \medskip
        \textbf{Implication:}\\
        SSIV is a weighted average of industry-level IVs with
        \textbf{Rotemberg weights} $\alpha_n$:
        \[
          \hat\beta_{\text{SSIV}} = \textstyle\sum_n \alpha_n\, \hat\beta_n
        \]

        \medskip
        \textbf{Test:} are shares correlated with pre-trends in outcome?
        Are Rotemberg weights non-negative?
      \end{block}
    \end{column}
  \end{columns}

  \medskip
  \begin{alertblock}{Two views — one instrument}
    Which view you take determines what you must defend. BHJ requires
    many quasi-randomly assigned shocks. GPSS requires clean historical
    shares. Both require the exclusion restriction.
  \end{alertblock}

\end{frame}

% --- Frame 11: Rotemberg weights ----------------------------
\begin{frame}{Rotemberg weights: which industries drive the IV?}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{What Rotemberg weights tell you}\\[4pt]
      The SSIV estimator is a weighted average of industry-specific
      IV estimates:
      \[
        \hat\beta_{\text{SSIV}} = \sum_n \alpha_n \hat\beta_n,
        \quad \sum_n \alpha_n = 1
      \]
      Industry $n$'s Rotemberg weight:
      \[
        \alpha_n \propto
        \left(\sum_\ell s_{\ell n} g_n\right)
        \cdot \text{(first-stage coeff.\ of industry $n$)}
      \]

      \medskip
      \textbf{Diagnostics}
      \begin{itemize}
        \item Which industries have the largest $|\alpha_n|$?
        \item Are any $\alpha_n < 0$? (negative weights signal fragility)
        \item Does $\hat\beta$ change sign when top industries are dropped?
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{ADH (2013) China shock example}

      \smallskip
      \begin{center}\scriptsize
      \renewcommand{\arraystretch}{1.3}
      \begin{tabular}{lrc}
        \toprule
        Industry & $\alpha_n$ & Sign \\
        \midrule
        Apparel \& textiles & 0.31 & $+$ \\
        Computer hardware & 0.23 & $+$ \\
        Furniture & 0.14 & $+$ \\
        Misc.\ manufacturing & 0.09 & $+$ \\
        Other (49 industries) & 0.23 & $+$ \\
        \bottomrule
      \end{tabular}
      \end{center}

      \smallskip
      Chinese apparel and computer imports drive $\approx$54\% of the
      China shock IV — concentrated but all positive.

      \medskip
      \begin{block}{\small Sensitivity check}
        \small Drop top 3 industries; re-estimate. If $\hat\beta$ is
        stable $\Rightarrow$ robust. If it flips $\Rightarrow$ fragile.
      \end{block}
    \end{column}
  \end{columns}

\end{frame}

% --- Frame 12: Practical considerations ---------------------
\begin{frame}[shrink=5]{SSIV in practice: three things you must do}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{1.\ Leave-one-out shocks}
        Exclude the focal region from the national shock computation
        to avoid mechanical correlation:
        \[
          g_n^{(-\ell)} = \frac{\Delta E_n - \Delta E_{\ell n}}{E_n - E_{\ell n}}
        \]
        Prevents spurious first stages driven by own-region variation.
      \end{block}

      \medskip
      \begin{block}{2.\ Cluster at shock level}
        Standard errors must account for shock-level correlation
        \citep{adao2019}.\\[2pt]
        Cluster at the \textbf{industry} (shock), not the
        local unit — industries induce correlated residuals across
        regions with similar shares.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{3.\ Incomplete shares}
        If $\sum_n s_{\ell n} < 1$ (e.g.\ only manufacturing covered):
        \begin{itemize}\small
          \item Normalise shares to sum to 1, \textbf{or}
          \item Include only covered industries and control for
                total employment share
        \end{itemize}
        Use \texttt{ssaggregate} (Stata) to handle this cleanly.
      \end{block}

      \medskip
      \begin{alertblock}{When SSIV fails}
        \begin{itemize}\small
          \item Few shocks + one industry dominates: weak instrument
          \item Shares reflect local sorting: GPSS assumption violated
          \item Shocks have economy-wide spillovers: general equilibrium
                effects contaminate the exclusion restriction
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}

\end{frame}
